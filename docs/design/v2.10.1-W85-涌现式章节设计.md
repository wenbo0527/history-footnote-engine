## 一、设计目标

将项目从"**预设分支图+玩家选路**"（嵌入叙事）升级到"**玩家行为涌现+DM 即时生成**"（涌现叙事）：

1. **玩家自由度提升**：玩家可做出设计者完全没想到的选择
2. **章节动态生成**：不再预设完整章节图，而是用模板+参数实时生成
3. **三重收束保证**：历史铁轨（硬）+ 模板序列（软）+ 价值收敛（软）
4. **判断分工清晰**：规则引擎判"变道"+ DM 判"具体叙事"

### 2.2 核心组件：RouteDetector

**文件**：`src/history_footnote/chapter/route_detector.py`（新建）

```python
"""🆕 v2.10.1 W85: 简单版路线检测器

Phase 1: 纯规则版（关键词 + 价值偏移阈值）
Phase 2: 加 LLM 意图分类（见 design section 4）

判断逻辑：
1. 关键词匹配 → 立即触发对应模板
2. 价值偏移 > 阈值 → 触发 rising_conflict
3. 历史铁轨触达 → 强制 convergence
4. 都未触发 → return False（保持当前路线）
"""
from __future__ import annotations
import re
from typing import Optional

from history_footnote.chapter.types import ChapterBlueprint


# 关键词路由表（可被 era_config 覆盖）
DEFAULT_ROUTE_KEYWORDS: dict[str, list[str]] = {
    "rising_conflict": [
        # 中文关键词
        "抗税", "不交", "逃税", "投奔", "逃跑", "告官", "反抗",
        "海瑞", "衙门", "闹事", "罢工", "起义",
        # 英文（如适用）
        "rebel", "flee", "protest",
    ],
    "crisis": [
        "倭寇", "灾难", "火灾", "洪水", "地震", "海瑞去世",
        "银税", "抄家", "破产", "死亡",
        "disaster", "death",
    ],
    "convergence": [
        "万历", "皇帝", "朝廷", "诏书", "圣旨",
    ],
}

# 价值偏移阈值
VALUE_SHIFT_THRESHOLD = 0.7  # 绝对值超过 0.7 触发路线变更

# 触发历史铁轨的关键词（决定强制 convergence）
HISTORICAL_ANCHOR_KEYWORDS = [
    "倭寇来袭", "万历驾崩", "海瑞去世", "银税改革", "一条鞭法",
]


class RouteDetector:
    """🆕 W85: 路线检测器（Phase 1 纯规则版）"""

    def __init__(self, route_keywords: Optional[dict[str, list[str]]] = None):
        self.route_keywords = route_keywords or DEFAULT_ROUTE_KEYWORDS

    def detect(
        self,
        player_input: str,
        value_shifts: dict[str, float],
        current_chapter: ChapterBlueprint,
        historical_anchors_triggered: list[str] | None = None,
    ) -> dict:
        """检测玩家行为是否构成新路线

        Args:
            player_input: 玩家本回合输入原文
            value_shifts: 当前 value_shifts dict（如 {"trust": -0.8, ...}）
            current_chapter: 当前章节蓝图
            historical_anchors_triggered: 本回合触发的历史铁轨列表

        Returns:
            {
                "route_change": bool,           # 是否触发新路线
                "suggested_template": str,       # 建议的模板（"rising_conflict" 等）
                "trigger": str,                  # 触发原因（"keyword:抗税" 等）
                "confidence": float,             # 置信度 0-1
                "dm_instruction": str,           # 给 DM 的创作指令
            }
        """

        # 1. 优先级最高：历史铁轨触达 → 强制 convergence
        if historical_anchors_triggered:
            anchor = historical_anchors_triggered[0]
            return {
                "route_change": True,
                "suggested_template": "convergence",
                "trigger": f"historical_anchor:{anchor}",
                "confidence": 1.0,
                "dm_instruction": (
                    f"历史铁轨事件「{anchor}」已触发。"
                    f"无论玩家之前做了什么路线，都必须在此刻汇合。"
                    f"根据玩家之前的选择（可参考 value_state），调整叙事入口。"
                ),
            }

        # 2. 关键词匹配
        keyword_match = self._match_keywords(player_input)
        if keyword_match:
            template, kw = keyword_match
            return {
                "route_change": True,
                "suggested_template": template,
                "trigger": f"keyword:{kw}",
                "confidence": 0.85,
                "dm_instruction": self._build_dm_instruction(
                    template, current_chapter, f"玩家行为触发关键词「{kw}」"
                ),
            }

        # 3. 价值偏移检测
        for dim, val in value_shifts.items():
            if abs(val) >= VALUE_SHIFT_THRESHOLD:
                return {
                    "route_change": True,
                    "suggested_template": "rising_conflict",
                    "trigger": f"value_shift:{dim}={val:.2f}",
                    "confidence": 0.7,
                    "dm_instruction": self._build_dm_instruction(
                        "rising_conflict", current_chapter,
                        f"玩家在「{dim}」维度上发生剧烈偏移（{val:+.2f}）"
                    ),
                }

        # 4. 未触发
        return {
            "route_change": False,
            "suggested_template": current_chapter.narrative_position,
            "trigger": None,
            "confidence": 0.0,
            "dm_instruction": "",
        }

    def _match_keywords(self, text: str) -> Optional[tuple[str, str]]:
        """返回 (template, keyword) 或 None"""
        if not text:
            return None
        for template, keywords in self.route_keywords.items():
            for kw in keywords:
                if kw in text:
                    return (template, kw)
        return None

    def _build_dm_instruction(
        self, template: str, current: ChapterBlueprint, reason: str
    ) -> str:
        """根据模板生成 DM 创作指令"""
        instructions = {
            "opening": "铺陈日常，暗示即将到来的变故",
            "rising_conflict": "冲突升级，逼迫玩家表态",
            "crisis": "不可逆事件发生，玩家被迫重新定位",
            "convergence": "历史铁轨事件落地，根据玩家路线调整叙事入口",
            "resolution": "收束所有线索，对照开头的日常",
        }
        base = instructions.get(template, "")
        return f"[{template}] {base}。\n触发原因：{reason}。\n当前章节：{current.chapter_title or '未命名'}。"
```

```python
class ChapterCoordinator:
    # ... 现有方法 ...

    def detect_route_change(
        self,
        player_input: str,
        value_shifts: dict[str, float],
        historical_anchors_triggered: list[str] | None = None,
    ) -> dict:
        """🆕 W85: 调 RouteDetector 检测路线变化

        Returns:
            RouteDetector.detect() 的结果 dict
        """
        from history_footnote.chapter.route_detector import RouteDetector
        detector = RouteDetector()
        cs = self.state.chapter_state
        # current_chapter 简化：直接从 cs.blueprint 取
        current = cs.blueprint or {}
        return detector.detect(
            player_input=player_input,
            value_shifts=value_shifts,
            current_chapter=current,
            historical_anchors_triggered=historical_anchors_triggered,
        )

    def apply_route_change(self, detection: dict) -> None:
        """🆕 W85: 应用路线变更到 state

        1. 写 current_route
        2. 追加 route_history
        3. 不立即改章节（让 DM 先按新路线创作，下一节点再结算）
        """
        if not detection.get("route_change"):
            return
        cs = self.state.chapter_state
        new_template = detection["suggested_template"]
        cs.current_route = {
            "template": new_template,
            "trigger": detection["trigger"],
            "entered_at_round": self.state.round_number,
            "dm_instruction": detection["dm_instruction"],
        }
        cs.route_history.append({
            "round": self.state.round_number,
            "from_template": cs.blueprint.get("narrative_position", "opening") if cs.blueprint else "opening",
            "to_template": new_template,
            "trigger": detection["trigger"],
        })
        _LOG.info(
            "[W85] 路线变更: round=%d, %s -> %s, trigger=%s",
            self.state.round_number,
            cs.route_history[-1]["from_template"],
            new_template,
            detection["trigger"],
        )
```

### 3.2 LLM 调用成本

- **每回合 1 次**（仅在 Phase 1 未触发时）
- **平均成功率 ~30%**（80% 走 Phase 1 关键词 + 价值）
- **额外 LLM 调用 ~30% × 1 次/回合 = 0.3 次/回合**
- 假设每回合 60s LLM → 增加 ~18s
- **可接受**

### 3.3 LLM Prompt 必须简短

避免 prompt 过长（> 500 tokens），仅传必要字段。

**目标**：最接近人类 DM 的即兴能力

### 4.1 双判断架构

```
RouteDetector 判"是否变道"（确定性）
   ↓
   是 → DM 调一次 LLM 判"变道后的具体叙事"
   ↓
   DM 判结果 + 规则引擎收束检查（避免无限发散）
   ↓
   最终结果写入 current_route
```

### 4.3 收束检查

**关键风险**：DM 可能判"变道"到完全不存在的路线

**收束检查**：
```python
def _convergence_check(self, suggested_template, must_resolve):
    """检查新模板是否合理"""
    if suggested_template not in NARRATIVE_POSITIONS:
        return False, f"Unknown template: {suggested_template}"

    # 不能从 opening 直接跳 resolution
    flow = ["opening", "rising_conflict", "crisis", "convergence", "resolution"]
    current_idx = flow.index(self.current_position)
    suggested_idx = flow.index(suggested_template)
    if suggested_idx < current_idx - 1:
        return False, f"不能倒退: {self.current_position} -> {suggested_template}"

    # 必须有 must_resolve
    if not must_resolve:
        return False, "no must_resolve provided"

    return True, "ok"
```
